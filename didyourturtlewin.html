            from turtle import Turtle, Screen
            import random

            is_race_on = False
            screen = Screen()

            screen.setup(width=500, height=400)
            user_bet = screen.textinput(title="Make your bet", prompt="Which turtle will win the race? Enter a color: ")
            colors = ["red", "orange", "yellow", "blue", "green", "purple"]
            y_position = [-70, -40, -10, 20, 50, 80]
            all_turtles = []

            for t in range(0, 6):
                new_turtle = Turtle(shape="turtle")
                new_turtle.penup()
                new_turtle.color(colors[t])
                # timmy.goto(x=-230, y=-100)
                #this was used before assigning diff turtles
                # x= 500/2= 250 that will go beyond the screen and thus -230 and y = 400/2=200 but we have to shift it to a particular
                # position not at center thus -100
                new_turtle.goto(x=-230, y=y_position[t])
                all_turtles.append(new_turtle)

            if user_bet:
                is_race_on = True

            while is_race_on:
                for turtle in all_turtles:
                    #250-20=230 bcoz turtle is 40 pixel wide thus we will subtract half value of the turtle
                    if turtle.xcor() > 230:
                        is_race_on = False
                        winning_color = turtle.pencolor()
                        if winning_color == user_bet:
                            print(f"You've won! The {winning_color} turtle is the winner.")
                        else:
                            print(f"You've lost! The {winning_color} turtle is the winner.")
                    rand_distace = random.randint(0, 10)
                    turtle.forward(rand_distace)

            screen.exitonclick()
